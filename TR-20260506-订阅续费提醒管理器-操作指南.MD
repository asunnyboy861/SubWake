- 自动识别准确率低

```swift
    case monthly = "Monthly"
    case quarterly = "Quarterly"

    case annually = "Annually"

        switch self {
        case .weekly: return .weekOfYear
        case .monthly: return .month
        case .quarterly: return .month
        case .semiAnnually: return .month
        case .annually: return .year

        case .monthly: return 1
        case .quarterly: return 3
```

### 4.3 智能提醒引擎核心代码

```swift
import UserNotifications
import Foundation

final class ReminderEngine {
    static let shared = ReminderEngine()
    
    private let notificationCenter = UNUserNotificationCenter.current()
    
    func requestAuthorization() async throws -> Bool {
        try await notificationCenter.requestAuthorization(options: [.alert, .sound, .badge])
    }
    
    func scheduleReminders(for subscription: Subscription) async {
        let content = UNMutableNotificationContent()
        content.title = "⏰ \(subscription.serviceName) Renewing Soon"
        content.body = String(
            format: "$%@ charge in %d days on %@",
            subscription.price.description,
            0,
            subscription.nextRenewalDate.formatted(date: .abbreviated, time: .omitted)
        )
        content.sound = .default
        content.categoryIdentifier = "SUBSCRIPTION_REMINDER"
        content.userInfo = [
            "subscriptionId": subscription.id.uuidString,
            "price": subscription.price.description,
            "serviceName": subscription.serviceName
        ]
        
        let actions = [
            UNNotificationAction(identifier: "KEEP_ACTION", title: "Keep It", options: []),
            UNNotificationAction(identifier: "REMIND_LATER", title: "Remind Tomorrow", options: []),
            UNNotificationAction(identifier: "VIEW_ACTION", title: "View Details", options: [.foreground])
        ]
        let category = UNNotificationCategory(
            identifier: "SUBSCRIPTION_REMINDER",
            actions: actions,
            intentIdentifiers: [],
            options: .customDismissAction
        )
        notificationCenter.setNotificationCategories([category])
        
        for daysBefore in subscription.reminderDaysBefore {
            guard let triggerDate = Calendar.current.date(
                byAdding: .day,
                value: -daysBefore,
                to: subscription.nextRenewalDate
            ) else { continue }
            
            let triggerComponents = Calendar.current.dateComponents(
                [.year, .month, .day, .hour, .minute],
                from: triggerDate
            )
            let trigger = UNCalendarNotificationTrigger(
                dateMatching: triggerComponents,
                repeats: false
            )
            
            let request = UNNotificationRequest(
                identifier: "\(subscription.id.uuidString)_\(daysBefore)",
                content: content,
                trigger: trigger
            )
            
            try? await notificationCenter.add(request)
        }
    }
    
    func scheduleTrialReminder(for subscription: Subscription) async {
        guard subscription.isTrial, let trialEnd = subscription.trialEndDate else { return }
        
        let content = UNMutableNotificationContent()
        content.title = "🚨 Free Trial Ending: \(subscription.serviceName)"
        content.body = "Your free trial ends in 3 days. $\(subscription.price) will be charged on \(trialEnd.formatted(date: .abbreviated, time: .omitted))."
        content.sound = .default
        content.categoryIdentifier = "TRIAL_REMINDER"
        content.userInfo = [
            "subscriptionId": subscription.id.uuidString,
            "isTrial": true
        ]
        
        let trialActions = [
            UNNotificationAction(identifier: "CANCEL_NOW", title: "Set Cancel Reminder", options: []),
            UNNotificationAction(identifier: "KEEP_TRIAL", title: "Keep After Trial", options: []),
            UNNotificationAction(identifier: "VIEW_ACTION", title: "View Details", options: [.foreground])
        ]
        let category = UNNotificationCategory(
            identifier: "TRIAL_REMINDER",
            actions: trialActions,
            intentIdentifiers: [],
            options: .customDismissAction
        )
        notificationCenter.setNotificationCategories([category])
        
        for daysBefore in [7, 3, 1] {
            guard let triggerDate = Calendar.current.date(
                byAdding: .day, value: -daysBefore, to: trialEnd
            ), triggerDate > Date() else { continue }
            
            let components = Calendar.current.dateComponents(
                [.year, .month, .day, .hour, .minute], from: triggerDate
            )
            let trigger = UNCalendarNotificationTrigger(
                dateMatching: components, repeats: false
            )
            let request = UNNotificationRequest(
                identifier: "trial_\(subscription.id.uuidString)_\(daysBefore)",
                content: content, trigger: trigger
            )
            try? await notificationCenter.add(request)
        }
    }
    
    func scheduleCancellationDeadlineReminder(for subscription: Subscription) async {
        guard let deadline = subscription.cancellationDeadline else { return }
        
        let content = UNMutableNotificationContent()
        content.title = "⚠️ Cancel Deadline: \(subscription.serviceName)"
        content.body = "You must cancel by \(deadline.formatted(date: .abbreviated, time: .omitted)) to avoid the next charge."
        content.sound = .default
        content.categoryIdentifier = "CANCEL_DEADLINE"
        
        for daysBefore in [14, 7, 3, 1] {
            guard let triggerDate = Calendar.current.date(
                byAdding: .day, value: -daysBefore, to: deadline
            ), triggerDate > Date() else { continue }
            
            let components = Calendar.current.dateComponents(
                [.year, .month, .day, .hour, .minute], from: triggerDate
            )
            let trigger = UNCalendarNotificationTrigger(
                dateMatching: components, repeats: false
            )
            let request = UNNotificationRequest(
                identifier: "deadline_\(subscription.id.uuidString)_\(daysBefore)",
                content: content, trigger: trigger
            )
            try? await notificationCenter.add(request)
        }
    }
    
    func cancelAllReminders(for subscriptionId: UUID) {
        let pending = notificationCenter.pendingNotificationRequests()
        Task {
            let requests = await pending
            let idsToRemove = requests
                .map(\.identifier)
                .filter { $0.contains(subscriptionId.uuidString) }
            notificationCenter.removePendingNotificationRequests(withIdentifying: idsToRemove)
        }
    }
    
    func refreshAllReminders(subscriptions: [Subscription]) async {
        notificationCenter.removeAllPendingNotificationRequests()
        for sub in subscriptions where sub.isActive {
            await scheduleReminders(for: sub)
            if sub.isTrial { await scheduleTrialReminder(for: sub) }
            if sub.cancellationDeadline != nil { await scheduleCancellationDeadlineReminder(for: sub) }
        }
    }
}
```

### 4.4 Widget实现核心代码

```swift
import WidgetKit
import SwiftUI
import SwiftData

struct SubWakeEntry: TimelineEntry {
    let date: Date
    let upcomingSubscriptions: [SubscriptionWidgetData]
    let monthlyTotal: Decimal
    let yearlyTotal: Decimal
}

struct SubscriptionWidgetData: Identifiable {
    let id: UUID
    let name: String
    let price: Decimal
    let daysUntilRenewal: Int
    let isTrial: Bool
    let colorHex: String
}

struct SubWakeWidget: Widget {
    let kind: String = "SubWakeWidget"
    
    var body: some WidgetConfiguration {
        StaticConfiguration(kind: kind, provider: SubWakeProvider()) { entry in
            SubWakeWidgetEntryView(entry: entry)
                .containerBackground(for: .widget) {
                    Color.black
                }
        }
        .configurationDisplayName("SubWake")
        .description("See upcoming subscription renewals at a glance.")
        .supportedFamilies([.systemSmall, .systemMedium, .systemLarge])
    }
}

struct SubWakeProvider: TimelineProvider {
    func placeholder(in context: Context) -> SubWakeEntry {
        SubWakeEntry(date: Date(), upcomingSubscriptions: [], monthlyTotal: 0, yearlyTotal: 0)
    }
    
    func getSnapshot(in context: Context, completion: @escaping (SubWakeEntry) -> Void) {
        let entry = SubWakeEntry(date: Date(), upcomingSubscriptions: [], monthlyTotal: 0, yearlyTotal: 0)
        completion(entry)
    }
    
    func getTimeline(in context: Context, completion: @escaping (Timeline<SubWakeEntry>) -> Void) {
        Task {
            let container = try! ModelContainer(for: Subscription.self)
            let context = ModelContext(container)
            let descriptor = FetchDescriptor<Subscription>(
                predicate: #Predicate { $0.isActive },
                sortBy: [SortDescriptor(\.nextRenewalDate)]
            )
            let subs = try context.fetch(descriptor)
            
            let widgetData = subs.prefix(5).map { sub in
                SubscriptionWidgetData(
                    id: sub.id,
                    name: sub.serviceName,
                    price: sub.price,
                    daysUntilRenewal: Calendar.current.dateComponents([.day], from: Date(), to: sub.nextRenewalDate).day ?? 0,
                    isTrial: sub.isTrial,
                    colorHex: sub.colorHex
                )
            }
            
            let monthlyTotal = subs.reduce(Decimal(0)) { $0 + $1.billingCycle.monthlyEquivalent }
            let yearlyTotal = monthlyTotal * 12
            
            let entry = SubWakeEntry(
                date: Date(),
                upcomingSubscriptions: widgetData,
                monthlyTotal: monthlyTotal,
                yearlyTotal: yearlyTotal
            )
            
            let timeline = Timeline(entries: [entry], policy: .atEnd)
            completion(timeline)
        }
    }
}
```

```swift
import MessageUI

final class EmailSubscriptionScanner {
    
    struct ScannedSubscription {
        let serviceName: String
        let price: Decimal
        let billingCycle: BillingCycle
        let renewalDate: Date?
        let isTrial: Bool
        let source: String
    }
    
    func scanEmailPatterns(in subject: String, body: String, sender: String) -> ScannedSubscription? {
        let trialPatterns = [
            "free trial",
            "trial period",
            "trial ends",
            "start your free"
        ]
        
        let renewalPatterns = [
            "renewal",
            "will be charged",
            "auto-renew",
            "subscription will renew",
            "upcoming payment",
            "billing date"
        ]
        
        let pricePattern = try? NSRegularExpression(
            pattern: "\\$([0-9]+\\.?[0-9]*)",
            options: .caseInsensitive
        )
        
        let isTrial = trialPatterns.contains { subject.lowercased().contains($0) || body.lowercased().contains($0) }
        let isRenewal = renewalPatterns.contains { subject.lowercased().contains($0) || body.lowercased().contains($0) }
        
        guard isTrial || isRenewal else { return nil }
        
        var price: Decimal = 0
        if let match = pricePattern?.firstMatch(in: body, range: NSRange(body.startIndex..., in: body)),
           let range = Range(match.range(at: 1), in: body),
           let parsedPrice = Decimal(string: String(body[range])) {
            price = parsedPrice
        }
        
        let serviceName = extractServiceName(from: sender)
        
        return ScannedSubscription(
            serviceName: serviceName,
            price: price,
            billingCycle: .monthly,
            renewalDate: extractDate(from: body),
            isTrial: isTrial,
            source: "Email Scan"
        )
    }
    
    private func extractServiceName(from sender: String) -> String {
        let knownServices: [String: String] = [
            "netflix": "Netflix",
            "spotify": "Spotify",
            "apple": "Apple",
            "google": "Google",
            "amazon": "Amazon",
            "hulu": "Hulu",
            "disney": "Disney+",
            "youtube": "YouTube Premium",
            "microsoft": "Microsoft 365",
            "adobe": "Adobe",
            "notion": "Notion",
            "figma": "Figma",
            "github": "GitHub",
            "medium": "Medium",
            "nytimes": "NY Times",
            "washingtonpost": "Washington Post"
        ]
        
        let lower = sender.lowercased()
        for (key, value) in knownServices {
            if lower.contains(key) { return value }
        }
        return sender.components(separatedBy: "@").first ?? sender
    }
    
    private func extractDate(from text: String) -> Date? {
        let detectors: [(NSDataDetector, Date)] = []
        guard let detector = try? NSDataDetector(types: NSTextCheckingResult.CheckingType.date.rawValue) else { return nil }
        let range = NSRange(text.startIndex..., in: text)
        let matches = detector.matches(in: text, range: range)
        return matches.first.flatMap { Range($0.range, in: text).map { text[$0] } }
            .flatMap { DateFormatter().date(from: String($0)) }
    }
}
```

5. **无API成本**: 纯本地运行，无服务器/API费用，一次性买断可持续

## 十一、核心差异化功能详解

### 11.1 智能分级提醒系统 (独有)

```
提醒分级逻辑:
══════════════

免费试用 → 7天前 + 3天前 + 1天前 (3次提醒，最紧急)
年度订阅 → 14天前 + 7天前 + 3天前 + 1天前 (4次提醒)
月度订阅 → 3天前 + 1天前 (2次提醒)
取消截止日 → 14天前 + 7天前 + 3天前 + 1天前 (4次提醒)
价格变动 → 立即通知 (1次)

用户可自定义每个订阅的提醒时间和频率
```

### 11.2 取消截止日提醒 (独家)

针对健身房、保险等需要提前N天通知取消的服务：
- 用户输入取消通知期(如30天)
- 系统自动计算取消截止日
- 在截止日前多次提醒
- 直接链接到App Store订阅管理页面

### 11.3 价格变动监控 (独家)

- 用户记录订阅价格
- 当价格变化时(用户手动更新)，记录历史价格
- 显示价格变动趋势
- 价格上涨时触发额外提醒

### 11.4 隐私优先架构

```
数据存储策略:
══════════════

✅ 所有数据存储在设备本地 (SwiftData)
✅ iCloud同步仅通过Apple CloudKit (端到端加密)
✅ 零第三方数据收集
✅ 零分析追踪
✅ 零广告
✅ 无需注册账号
✅ 无需连接银行
✅ App Transport Security强制HTTPS
✅ 数据导出功能(CSV) - 用户完全控制自己的数据

隐私标签(App Store):
- 数据收集: 无
- 数据关联: 无
- 数据使用: 无
```
